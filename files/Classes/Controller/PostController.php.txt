use ONEANDONE\Board\Utility\Breadcrumb;

/**
 * PostController
 */
class PostController extends \TYPO3\CMS\Extbase\Mvc\Controller\ActionController
{

    /**
     * postRepository
     *
     * @var \ONEANDONE\Board\Domain\Repository\PostRepository
     * @inject
     */
    protected $postRepository = NULL;

    /**
     * user
     *
     * @var \ONEANDONE\Board\Utility\User
     * @inject
     */
    protected $user = NULL;

    /**
     * breadcrumb
     *
     * @var \ONEANDONE\Board\Utility\Breadcrumb
     * @inject
     */
    protected $breadcrumb = NULL;

    /**
     *
     */
    public function initializeView()
    {

        if ($this->user->loggedIn()) {
            $this->view->assign('user', $this->user);
        }

        $this->view->assign('breadcrumb', $this->breadcrumb);

    }

    /**
     * action new
     *
     * @return void
     */
    public function newAction()
    {

    }

    /**
     * action create
     *
     * @param \ONEANDONE\Board\Domain\Model\Post $newPost
     * @return void
     */
    public function createAction(\ONEANDONE\Board\Domain\Model\Post $newPost)
    {

        $newPost->setCrdate(time());
        $this->postRepository->add($newPost);

        $this->redirect('show', 'Topic', 'Board', [
            'topic' => $newPost->getTopicid()
        ]);

    }

    /**
     * action edit
     *
     * @param \ONEANDONE\Board\Domain\Model\Post $post
     * @ignorevalidation $post
     * @return void
     */
    public function editAction(\ONEANDONE\Board\Domain\Model\Post $post)
    {

        $post = $this->postRepository->findByUid($post->getUid());

        $this->breadcrumb->addElement('Post');
        $this->breadcrumb->addElement($post->getUid());
        $this->breadcrumb->addElement('Update');

        $this->view->assign('post', $post);
    }

    /**
     * action update
     *
     * @param \ONEANDONE\Board\Domain\Model\Post $post
     * @return void
     */
    public function updateAction(\ONEANDONE\Board\Domain\Model\Post $post)
    {

        $this->postRepository->update($post);

        $this->redirect('show', 'Topic', 'Board', [
            'topic' => $post->getTopicid()
        ]);

    }

    /**
     * action delete
     *
     * @param \ONEANDONE\Board\Domain\Model\Post $post
     * @return void
     */
    public function deleteAction(\ONEANDONE\Board\Domain\Model\Post $post)
    {
        $this->addFlashMessage('The object was deleted. Please be aware that this action is publicly accessible unless you implement an access check. See http://wiki.typo3.org/T3Doc/Extension_Builder/Using_the_Extension_Builder#1._Model_the_domain', '', \TYPO3\CMS\Core\Messaging\AbstractMessage::ERROR);
        $this->postRepository->remove($post);
        $this->redirect('list');
    }

}
