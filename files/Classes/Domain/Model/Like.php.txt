/**
 * Like
 */
class Like extends \TYPO3\CMS\Extbase\DomainObject\AbstractEntity
{

    /**
     * authorid
     *
     * @var int
     * @validate NotEmpty
     */
    protected $authorid = 0;
    
    /**
     * postid
     *
     * @var int
     * @validate NotEmpty
     */
    protected $postid = 0;
    
    /**
     * Returns the authorid
     *
     * @return int $authorid
     */
    public function getAuthorid()
    {
        return $this->authorid;
    }
    
    /**
     * Sets the authorid
     *
     * @param int $authorid
     * @return void
     */
    public function setAuthorid($authorid)
    {
        $this->authorid = $authorid;
    }
    
    /**
     * Returns the postid
     *
     * @return int $postid
     */
    public function getPostid()
    {
        return $this->postid;
    }
    
    /**
     * Sets the postid
     *
     * @param int $postid
     * @return void
     */
    public function setPostid($postid)
    {
        $this->postid = $postid;
    }

}
