/**
 * BoardController
 */
class Markdown
{

    public function render($text)
    {

        $text = preg_replace('#\#\#\#(.+?)\#\#\##', '<h3>\\1</h3>', $text);
        $text = preg_replace('#\#\#(.+?)\#\##', '<h2>\\1</h2>', $text);
        $text = preg_replace('#\#(.+?)\##', '<h1>\\1</h1>', $text);
        $text = preg_replace('#\[code\](.+?)\[/code\]#', '<pre><code>\\1</code></pre>', $text);
        $text = preg_replace('#\[quote\](.+?)\[/quote\]#', '<blockquote>\\1</blockquote>', $text);

        return $text;
        
    }

}
