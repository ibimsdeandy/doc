/**
 * Post
 */
class Post extends \TYPO3\CMS\Extbase\DomainObject\AbstractEntity
{

    /**
     * markdown
     * @var \ONEANDONE\Board\Utility\Markdown
     * @inject
     */
    protected $markdown = NULL;

    /**
     * author
     *
     * @var int
     * @validate NotEmpty
     */
    protected $author = 0;

    /**
     * crdate
     *
     * @var int
     */
    protected $crdate = 0;
    
    /**
     * topicid
     *
     * @var int
     * @validate NotEmpty
     */
    protected $topicid = 0;
    
    /**
     * text
     *
     * @var string
     */
    protected $text = '';

    /**
     * Returns the crdate
     *
     * @return int $crdate
     */
    public function getCrdate()
    {
        return $this->crdate;
    }

    /**
     * Sets the crdate
     *
     * @param int $crdate
     * @return void
     */
    public function setCrdate($crdate)
    {
        $this->crdate = $crdate;
    }
    
    /**
     * Returns the author
     *
     * @return int $author
     */
    public function getAuthor()
    {
        return $this->author;
    }
    
    /**
     * Sets the author
     *
     * @param int $author
     * @return void
     */
    public function setAuthor($author)
    {
        $this->author = $author;
    }
    
    /**
     * Returns the topicid
     *
     * @return int $topicid
     */
    public function getTopicid()
    {
        return $this->topicid;
    }
    
    /**
     * Sets the topicid
     *
     * @param int $topicid
     * @return void
     */
    public function setTopicid($topicid)
    {
        $this->topicid = $topicid;
    }
    
    /**
     * Returns the text
     *
     * @return string $text
     */
    public function getText()
    {
        return $this->markdown->render($this->text);
    }
    
    /**
     * Sets the text
     *
     * @param string $text
     * @return void
     */
    public function setText($text)
    {
        $this->text = $text;
    }

}
