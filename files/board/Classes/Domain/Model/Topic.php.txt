/**
 * Topic
 */
class Topic extends \TYPO3\CMS\Extbase\DomainObject\AbstractEntity
{

    /**
     * title
     *
     * @var string
     * @validate NotEmpty
     */
    protected $title = '';

    /**
     * password
     *
     * @var string
     */
    protected $password = '';

    /**
     * crdate
     *
     * @var int
     */
    protected $crdate = '';

    /**
     * text
     *
     * @var string
     * @validate NotEmpty
     */
    protected $text = '';

    /**
     * boardid
     *
     * @var int
     * @validate NotEmpty
     */
    protected $boardid = 0;

    /**
     * author
     *
     * @var int
     * @validate NotEmpty
     */
    protected $author = 0;

    /**
     * locked
     *
     * @var int
     */
    protected $locked = 0;

    /**
     * posts
     *
     * @var array
     */
    protected $posts = [];

    /**
     * board
     *
     * @var \ONEANDONE\Board\Domain\Model\Board
     */
    protected $board = NULL;

    /**
     * latest
     *
     * @var \ONEANDONE\Board\Domain\Model\Post
     */
    protected $latest = NULL;

    /**
     * Returns the title
     *
     * @return string $title
     */
    public function getTitle()
    {
        return $this->title;
    }

    /**
     * Sets the title
     *
     * @param string $title
     * @return void
     */
    public function setTitle($title)
    {
        $this->title = $title;
    }

    /**
     * Returns the locked
     *
     * @return int $locked
     */
    public function getLocked()
    {
        return $this->locked;
    }

    /**
     * Returns the locking state
     *
     * @return boolean $locked
     */
    public function getUnlocked()
    {

        $cookie = $_COOKIE['Topic_Unlocked_' . $this->getUid()];

        if ($cookie == 'true') {
            return true;
        }

        return false;
        
    }

    /**
     * Sets the locked
     *
     * @param int $locked
     * @return void
     */
    public function setLocked($locked)
    {
        $this->locked = $locked;
    }

    /**
     * Returns the crdate
     *
     * @return int $crdate
     */
    public function getCrdate()
    {
        return $this->crdate;
    }

    /**
     * Sets the crdate
     *
     * @param int $crdate
     * @return void
     */
    public function setCrdate($crdate)
    {
        $this->crdate = $crdate;
    }

    /**
     * Returns the text
     *
     * @return string $text
     */
    public function getText()
    {
        return $this->text;
    }

    /**
     * Sets the text
     *
     * @param string $text
     * @return void
     */
    public function setText($text)
    {
        $this->text = $text;
    }

    /**
     * Returns the password
     *
     * @return string $password
     */
    public function getPassword()
    {
        return $this->password;
    }

    /**
     * Sets the password
     *
     * @param string $password
     * @return void
     */
    public function setPassword($password)
    {
        $this->password = $password;
    }

    /**
     * Returns the posts
     *
     * @return string $posts
     */
    public function getPosts()
    {
        return $this->posts;
    }

    /**
     * Sets the posts
     *
     * @param string $posts
     * @return void
     */
    public function setPosts($posts)
    {
        $this->posts = $posts;
    }

    /**
     * Returns the boardid
     *
     * @return int $boardid
     */
    public function getBoardid()
    {
        return $this->boardid;
    }

    /**
     * Sets the boardid
     *
     * @param int $boardid
     * @return void
     */
    public function setBoardid($boardid)
    {
        $this->boardid = $boardid;
    }

    /**
     * Returns the board
     *
     * @return \ONEANDONE\Board\Domain\Model\ $board
     */
    public function getBoard()
    {
        return $this->board;
    }

    /**
     * Sets the board
     *
     * @param array $board
     * @return void
     */
    public function setBoard($board)
    {
        $this->board = $board;
    }

    /**
     * Returns the author
     *
     * @return int $author
     */
    public function getAuthor()
    {
        return $this->author;
    }

    /**
     * Sets the author
     *
     * @param int $author
     * @return void
     */
    public function setAuthor($author)
    {
        $this->author = $author;
    }

    /**
     * Returns the latest
     *
     * @return \ONEANDONE\Board\Domain\Model\Post $latest
     */
    public function getLatest()
    {
        return $this->latest;
    }

    /**
     * Sets the latest
     *
     * @param \ONEANDONE\Board\Domain\Model\Post $latest
     * @return void
     */
    public function setLatest($latest)
    {
        $this->latest = $latest;
    }

}
